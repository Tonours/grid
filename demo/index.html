Responsive proportional grid

Usage

SCSS

@import "grid";
@include grid; // Mobile-first
@media only screen and (min-width: 481px) {
  @include grid(tablet); // Tablet and up
}
@media only screen and (min-width: 769px) {
  @include grid(desktop); // Desktop and up
}

HTML

<div class="grid-wrapper">
  <div class="grid tablet-half desktop-third">
    Content
  </div>
  <div class="grid tablet-half desktop-two-thrids">
    Content
  </div>
</div>

"Semantic" grid

HTML

<ul class="medias">
  <li class="media-item">
    <img alt="" class="media-img" src="http://placekitten.com/320/240" />
  </li>
</ul>

SCSS

@import "grid";
$grid-selector: "%";
@include grid; // Mobile-first
@media only screen and (min-width: 481px) {
  @include grid(tablet); // Tablet and up
}
@media only screen and (min-width: 769px) {
  @include grid(desktop); // Desktop and up
}
.medias {
  @extend %grid-wrapper;
  overflow: hidden;
}
  .media-item {
    @extend %grid, %half, %tablet-third, %desktop-quarter;
  }
    .media-img {
      border: 0;
      display: block;
      width: 100%;
    }
